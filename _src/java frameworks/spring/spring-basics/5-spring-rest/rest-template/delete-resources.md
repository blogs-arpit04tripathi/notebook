```java
public void deleteSpittle(long id) {
  RestTemplate rest = new RestTemplate();
  rest.delete(URI.create("http://localhost:8080/spittr-api/spittles/" + id));
  rest.delete("http://localhost:8080/spittr-api/spittles/{id}", id));
}
```
